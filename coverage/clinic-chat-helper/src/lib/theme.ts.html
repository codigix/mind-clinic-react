// Utility function to get theme classes
export const getThemeClass = (path: string): string => {
  const keys = path.split('.');
  let result: any = theme;
  
  for (const key of keys) {
    if (result[key] === undefined) return '';
    result = result[key];
  }
  
  return result;
};
